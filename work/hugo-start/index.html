hugo 实践记录

安装

https://gohugo.io/installation/windows/直接用 scoop 安装最方便
go 安装msi文件运行就可以了

新建

  1. hugo new site [网站名字]

  2. 挑选一个主题 https://themes.gohugo.io/

  3. git submodule add [主题的github地址]

  4. 在 hugo.toml 里面定义主题 theme = '主题名字'

  5. hugo server 启动

  6. hugo server -D 预览

  7. 新建文章 hugo new content posts/my-first-post.md

  8. hugo 打包文件 会放到 public 目录下面

github page

  1. 创建一个仓库 命名 username.github.io
  2. 然后把远程仓库拉到本地 git clone 远程仓库
  3. 把上面 public 目录下的所有文件复制过来 提交
  4. 推送远程仓库
  5. 访问 username.github.io

新建一个文章

大部分文章我都不是直接通过 md 来写，而是先在语雀上写好，然后导出为 md 文档，同时保留语雀的换行但是这样导出的文章，有两个问题

  1. 换行符是跟在文本后面，并没有生效，我需要手动对换行符进行处理。
  2. 图片是 cdn 上的资源，我希望图片放在项目的静态资源文件夹里

所以需要一个程序来帮我对内容进行处理。

const fs = require("fs");
const path = require("path");
const axios = require("axios");

// 从命令行读取需要转换的 markdown 文件的地址
const markdownFilePath = process.argv[2];
// 从命令行读取转换之后的文件的路径
const outputFile = process.argv[3];
// 路径里获取没有文件后缀名的文件名
const fileName = process.argv[3].replace(/^(.*\/)?([^/.]+)\..+$/, '$2');

if (!markdownFilePath) {
  console.error("确认需要转换的 markdown 文件路径");
  process.exit(1);
}

fs.readFile(markdownFilePath, "utf-8", (err, data) => {
  if (err) {
    console.error("读错误", err);
    return;
  }
  const fileContent = transform(data);
  fs.appendFile(outputFile, fileContent, "utf-8", (err) => {
    if (err) {
      console.error("写入错误", err);
      return;
    }
  });
});

/**
 *
 * @param {string} data
 */
function transform(data) {
  // 转换换行符
  let contentWithoutBr = data.replace(/<br \/>/g, "    \n");
  const imageRegex = /\!\[[^\]]+\]\((https?:\/\/[^\s)]+)\)/g;
  let match;
  let num = 0;

  // 获取全部的语雀 cdn 图片链接 
  while ((match = imageRegex.exec(data)) !== null) {
    downLoadImages(match[1],`${fileName}-${num}.png`);
    // 把链接替换成下载下来的图片的名字
    contentWithoutBr = contentWithoutBr.replace(match[1], `/images/${fileName}-${num}.png`);
    num++;
  }
  return contentWithoutBr;
}


async function downLoadImages(imageUrl, fileName) {
  // 发请求获取文件并保存到静态目录文件夹的图片下
    try {
        const response = await axios.get(imageUrl, {
            responseType: 'stream'
        });
        const writer = fs.createWriteStream(`./static/images/${fileName}`);
        response.data.pipe(writer);

        writer.on('error', (err) => {
            console.log(err);
            fs.unlink(`./${fileName}.png`, () => {})
        })
    }catch (err) {
        console.log('请求图片错误', err);
    }
}

执行上面的文件需要传递两个参数，需要转化的 markdown 文件的地址，和创建的文章所在的地址当然图方便也可以用脚本实现

if [-z "$1"]; then
    echo "请输入md文件地址"
    exit
fi

if [-z "$2"]; then
    echo "请输入输出文件的地址"
    exit
fi
